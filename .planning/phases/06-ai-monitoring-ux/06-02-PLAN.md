<objective>
Render GitHub-style "+N -N" line count badges on treemap file and directory rects when GitDiff mode is active.

Purpose: The existing GitDiff color overlay shows "how hot" (color intensity) but not "how much" (actual line counts). Adding +/- badges gives the user concrete change context without leaving the treemap.

Output: `draw_diff_badge()` function in `rects.rs`, `aggregate_dir_diff()` helper, wired into `draw_file_rect()` and `draw_section_rect()`. Badge renders bottom-right with green "+" and red "-" text, hidden on small rects and for zero-change files.

    fn aggregate_dir_diff(
        by_file: &std::collections::HashMap<String, crate::core::pmat_types::FileDiffData>,
        dir_prefix: &str,
    ) -> (u32, u32) {
        // If dir_prefix is empty, aggregate all. Otherwise filter by prefix.
        // Returns (total_added, total_removed).
        // Pattern: identical to aggregate_dir_delta filter logic.
        let iter: Box<dyn Iterator<Item = &crate::core::pmat_types::FileDiffData>> = if dir_prefix.is_empty() {
            Box::new(by_file.values())
        } else {
            Box::new(
                by_file.iter()
                    .filter(|(k, _)| k.starts_with(dir_prefix))
                    .map(|(_, v)| v)
            )
        };
        iter.fold((0u32, 0u32), |(a, r), d| (a + d.lines_added, r + d.lines_removed))
    }
    ```

    fn draw_diff_badge(painter: &egui::Painter, rect: egui::Rect, added: u32, removed: u32) {
        // Guard: skip if rect too small
        if rect.width() < 60.0 || rect.height() < 14.0 {
            return;
        }
        // Guard: skip +0 -0 (per CONTEXT.md Claude's discretion)
        if added == 0 && removed == 0 {
            return;
        }
        let green = egui::Color32::from_rgb(80, 200, 80);
        let red   = egui::Color32::from_rgb(220, 60, 60);
        let font  = egui::FontId::monospace(8.0);
        let y = rect.bottom() - 2.0;
        let mut x = rect.right() - 2.0;
        // Removed (red) drawn rightmost, added (green) to left of it
        if removed > 0 {
            let text = format!("-{}", removed);
            let w = text.len() as f32 * 5.0;
            painter.text(
                egui::pos2(x, y),
                egui::Align2::RIGHT_BOTTOM,
                &text,
                font.clone(),
                red,
            );
            x -= w + 3.0;
        }
        if added > 0 {
            let text = format!("+{}", added);
            painter.text(
                egui::pos2(x, y),
                egui::Align2::RIGHT_BOTTOM,
                &text,
                font.clone(),
                green,
            );
        }
    }
    ```
