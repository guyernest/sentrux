    **Rewrite `run_check` (currently at line ~102):**

    Old behavior: scans directory, runs metrics::compute_health, runs rules::check_rules against .sentrux/rules.toml, prints report, exits 0/1.

    New behavior: delegates to `pmat quality-gate`:
    ```rust
    fn run_check(path: &str) -> i32 {
        eprintln!("[check] Running PMAT quality gate on {path}");

        let status = std::process::Command::new("pmat")
            .args(["quality-gate", "--format", "json", "--fail-on-violation", "--path", path])
            .stdout(std::process::Stdio::inherit())  // let pmat output flow to user
            .stderr(std::process::Stdio::inherit())
            .status();

        match status {
            Ok(s) if s.success() => {
                eprintln!("[check] Quality gate passed");
                0
            }
            Ok(s) => {
                eprintln!("[check] Quality gate failed (exit {})", s.code().unwrap_or(-1));
                1
            }
            Err(e) => {
                eprintln!("[check] Failed to run pmat: {e}");
                eprintln!("[check] Install PMAT: cargo install pmat");
                1
            }
        }
    }
    ```

    **Rewrite `run_gate` (currently at line ~171):**

    Old behavior: scans directory, computes arch report, compares against .sentrux/baseline.json, optionally saves baseline.

    New behavior: delegates to `pmat tdg --min-grade`:
    ```rust
    fn run_gate(path: &str, save_mode: bool) -> i32 {
        if save_mode {
            // Save mode doesn't apply to PMAT gate — warn user
            eprintln!("[gate] Note: --save is not supported with PMAT gate. Use `pmat config` to set grade thresholds.");
        }

        eprintln!("[gate] Running PMAT TDG grade gate on {path}");

        // Default minimum grade: C (configurable in future)
        let status = std::process::Command::new("pmat")
            .args(["tdg", "--min-grade", "C", "--format", "json", "--path", path])
            .stdout(std::process::Stdio::inherit())
            .stderr(std::process::Stdio::inherit())
            .status();

        match status {
            Ok(s) if s.success() => {
                eprintln!("[gate] Grade gate passed");
                0
            }
            Ok(s) => {
                eprintln!("[gate] Grade gate failed (exit {})", s.code().unwrap_or(-1));
                1
            }
            Err(e) => {
                eprintln!("[gate] Failed to run pmat: {e}");
                eprintln!("[gate] Install PMAT: cargo install pmat");
                1
            }
        }
    }
    ```

    Remove all `use` imports from deleted metrics modules (HealthReport, ArchReport, compute_health, etc.) in main.rs.

    If `run_check`/`run_gate` currently call `scan_directory` before metrics, remove those scan calls — PMAT does its own scanning.

    Use `.expect()` instead of `.unwrap()` if any unwrap is needed (per Task 2 of Plan 01 pattern).
