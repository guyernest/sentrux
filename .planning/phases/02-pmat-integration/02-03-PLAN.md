    **1. Create `sentrux-core/src/app/panels/pmat_panel.rs`:**

    This panel serves two purposes:
    - **Health summary section:** Shows PMAT repo-score (total_score, grade, category breakdown)
    - **File detail section:** When a file is selected (state.selected_path), shows TDG component scores

    Structure:
    ```rust
    use egui::Ui;
    use crate::app::state::AppState;
    use crate::core::pmat_types::{grade_to_display, PmatReport};

    pub fn draw_pmat_panel(ui: &mut Ui, state: &AppState) {
        // If no pmat_report, show "PMAT analysis not available" message
        let Some(report) = &state.pmat_report else {
            ui.label("PMAT analysis not available");
            ui.label("Install: cargo install pmat");
            return;
        };

        // --- Health Summary ---
        draw_health_summary(ui, report);

        ui.separator();

        // --- File Detail (if file selected) ---
        if let Some(selected) = &state.selected_path {
            draw_file_detail(ui, report, selected);
        } else {
            ui.label("Select a file to see TDG breakdown");
        }
    }

    fn draw_health_summary(ui: &mut Ui, report: &PmatReport) {
        // Show TDG average
        ui.heading("Project Health");
        ui.horizontal(|ui| {
            ui.label("TDG Grade:");
            ui.strong(grade_to_display(&report.tdg.average_grade));
        });
        ui.horizontal(|ui| {
            ui.label("TDG Score:");
            ui.label(format!("{:.1}", report.tdg.average_score));
        });
        ui.horizontal(|ui| {
            ui.label("Files Analyzed:");
            ui.label(format!("{}", report.tdg.total_files));
        });

        // Repo score if available
        if let Some(repo) = &report.repo_score {
            ui.separator();
            ui.horizontal(|ui| {
                ui.label("Repo Health:");
                ui.strong(format!("{} ({:.0}/110)", grade_to_display(&repo.grade), repo.total_score));
            });

            // Category breakdown with collapsible
            egui::CollapsingHeader::new("Categories").show(ui, |ui| {
                for (name, cat) in &repo.categories {
                    ui.horizontal(|ui| {
                        let status_color = match cat.status.as_str() {
                            "Pass" => egui::Color32::GREEN,
                            "Warning" => egui::Color32::YELLOW,
                            _ => egui::Color32::RED,
                        };
                        ui.colored_label(status_color, "●");
                        ui.label(format!("{}: {:.0}/{:.0}", name, cat.score, cat.max_score));
                    });
                }
            });
        }
    }

    fn draw_file_detail(ui: &mut Ui, report: &PmatReport, path: &str) {
        // Look up file in pmat report
        let file_score = report.by_path.get(path)
            .and_then(|&idx| report.tdg.files.get(idx));

        let Some(score) = file_score else {
            ui.label("No TDG data for this file");
            return;
        };

        ui.heading("TDG Breakdown");
        ui.horizontal(|ui| {
            ui.label("Grade:");
            ui.strong(grade_to_display(&score.grade));
            ui.label(format!("({:.1}/100)", score.total));
        });

        // Component scores — progressive disclosure via collapsing header
        egui::CollapsingHeader::new("Component Scores").default_open(true).show(ui, |ui| {
            score_row(ui, "Structural Complexity", score.structural_complexity);
            score_row(ui, "Semantic Complexity", score.semantic_complexity);
            score_row(ui, "Duplication Ratio", score.duplication_ratio);
            score_row(ui, "Coupling Score", score.coupling_score);
            score_row(ui, "Doc Coverage", score.doc_coverage);
            score_row(ui, "Consistency", score.consistency_score);
            score_row(ui, "Entropy", score.entropy_score);
        });

        // Penalties if any
        if !score.penalties_applied.is_empty() {
            egui::CollapsingHeader::new("Penalties").show(ui, |ui| {
                for p in &score.penalties_applied {
                    ui.label(format!("-{:.1}: {} ({})", p.amount, p.issue, p.source_metric));
                }
            });
        }

        // Critical defects
        if score.has_critical_defects {
            ui.colored_label(egui::Color32::RED,
                format!("{} critical defect(s)", score.critical_defects_count));
        }
    }

    fn score_row(ui: &mut Ui, label: &str, value: f32) {
        ui.horizontal(|ui| {
            ui.label(format!("{label}:"));
            ui.label(format!("{value:.1}"));
        });
    }
    ```
